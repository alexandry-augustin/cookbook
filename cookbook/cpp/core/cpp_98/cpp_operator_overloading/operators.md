int a = 1;		// On commence avec a = 1
a++; 			// a++ ==> a = 2
int b = a++; 		// Postfix: b = a ==> b = 2 , puis a++ ==> a = 3
int c = ++a; 		// Prefix: ++a ==> a = 4 , puis c = a ==> c = 4

a -= b			a = a - b
